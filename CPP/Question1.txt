#include<iostream>
#include<string>
using namespace std;
int getLIndex(char string[100],char  c)
{
	int size = sizeof(string),i=0;
	while(i<size)
	{
		
		if(string[i]==c){
		    return i;
		}	
		i++; 
	}
	
}
int getFIndex(char string[100],char c)
{
	int size = sizeof(string);
	int i=size; 
	while(i>=0)
{		
		if(string[i]==c){
		    return i;
		}	
	    i--;
}
		
}
int main()
{
	char str[100];
	char ch; 
	int Lindex,Findex; 
	cin>>str;
	cin>>ch;
	Lindex = getLIndex(str,ch);
	Findex = getFIndex(str,ch);
	if(Lindex!=-1)
		cout<<Lindex<<endl<<Findex;
	else
		cout<<"NOT FOUND"<<endl;
	return 0;	
}
